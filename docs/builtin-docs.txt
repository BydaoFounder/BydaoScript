Пример встроенной документации на уровне языка BS.
Ключевое слово "doc" определяет блок информации, которая используется
для генерации документации и мета-данных модуля.
Каждый блок doc задается непосредственно перед элементом языка (переменной,
функцией, константой, классом и т.д.), к которому относится.
Можно использовать специальные типы документации - @fuc, @const и т.п.
Для модуля в целом - обязательно @module.

// Модуль
@module {
    "Системные функции ввода/вывода"
    version "1.0"
    author "Иван Петров"
}

// Константа (короткая форма, без @)
doc { "Максимальный размер буфера" }
const MAX_BUF = 1024

// Переменная
doc { "Текущий уровень логирования" }
var logLevel = 3

// Функция с подробным описанием
@func {
    "Парсит строку в целое число"
    detail "Поддерживает десятичные и шестнадцатеричные (0x) числа.
            При ошибке возвращает 0."
    args {
        str "Строка для парсинга"
        radix "Система счисления (10 или 16). По умолчанию 10"
    }
    return "Целое число или 0 при ошибке"
}
func parseInt(str: String, radix: Int = 10) : Int

// Функция с краткой документацией
@func {
    "Выводит строку в консоль"
    arg text "Строка для вывода"
}
func outln(text: String)

// Статический метод
@func {
    "Возвращает максимальное из двух чисел"
    args {
        a "Первое число"
        b "Второе число"
    }
    return "Максимальное значение"
}
static func max(a: Int, b: Int) : Int
