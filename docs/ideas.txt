1. Генератор числового диапазона:

enum Int.range(1,10) as val {
    Sys.outln( val )
}

вместо

var val = 1
while val <= 10 next val += 1 {
}

Итератор можно определить как класс

iter Range {

    prop readonly key
    prop readonly value

    var m_from
    var m_to
    var m_delta

    Range( from, to, delta = 1 ) {
        m_from = from
        m_to = to
        m_delta = delta;
    }

    func next() {
        if key == null {
            key = 0
            value = m_from
        }
        else if value < m_to {
            key += 1
            value += m_delta
        }
        if value >= m_to {
            return false
        }
        return true
    }

    func reset() {
        key = null
    }
}
